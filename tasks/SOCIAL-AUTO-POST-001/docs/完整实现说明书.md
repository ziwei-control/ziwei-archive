## 1. 产品概述

### **1.1 产品定位**

一个多平台社交媒体自动发布工具，支持闲鱼、转转、eBay 等二手交易平台，实现一键发布、定时发布、多账号管理。

## 2. 功能详解

### **2.1 核心功能**

#### **功能 1: 多平台一键发布**

**实现流程**:
```
准备商品信息
    ↓
选择发布平台
    ↓
自动填充表单
    ↓
上传图片
    ↓
提交发布
    ↓
确认结果
```

**代码实现** (`src/publisher.py`):
```python
#!/usr/bin/env python3
"""
多平台发布器
"""

from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC
import time

class MultiPlatformPublisher:
    """多平台发布器"""
    
    def __init__(self, platform: str, driver_path: str = None):
        """
        初始化发布器
        
        Args:
            platform: 平台名称 (xianyu/zhuanzhuan/ebay)
            driver_path: ChromeDriver 路径
        """
        self.platform = platform
        self.driver = None
        self.driver_path = driver_path or '/usr/bin/chromedriver'
        
    def start_browser(self):
        """启动浏览器"""
        options = webdriver.ChromeOptions()
        options.add_argument('--no-sandbox')
        options.add_argument('--disable-dev-shm-usage')
        options.add_argument('--disable-blink-features=AutomationControlled')
        
        self.driver = webdriver.Chrome(
            executable_path=self.driver_path,
            options=options
        )
        
    def publish_to_xianyu(self, item: dict) -> bool:
        """
        发布到闲鱼
        
        Args:
            item: 商品信息字典
            
        Returns:
            发布成功返回 True
        """
        try:
            # 1. 打开闲鱼发布页面
            self.driver.get('https://2.taobao.com/sell.htm')
            
            # 2. 等待页面加载
            time.sleep(3)
            
            # 3. 填写标题
            title_input = self.driver.find_element(By.NAME, 'title')
            title_input.send_keys(item['title'])
            
            # 4. 填写描述
            desc_input = self.driver.find_element(By.NAME, 'description')
            desc_input.send_keys(item['description'])
            
            # 5. 填写价格
            price_input = self.driver.find_element(By.NAME, 'price')
            price_input.send_keys(str(item['price']))
            
            # 6. 上传图片
            if 'images' in item:
                upload_input = self.driver.find_element(By.NAME, 'images')
                for image in item['images']:
                    upload_input.send_keys(image)
            
            # 7. 选择分类
            # ... (根据实际页面元素调整)
            
            # 8. 提交发布
            submit_btn = self.driver.find_element(By.XPATH, '//button[@type="submit"]')
            submit_btn.click()
            
            # 9. 等待发布完成
            time.sleep(5)
            
            # 10. 检查发布结果
            success = self._check_publish_success()
            
            return success
            
        except Exception as e:
            print(f"❌ 发布失败：{e}")
            return False
    
    def _check_publish_success(self) -> bool:
        """检查发布是否成功"""
        # 检查成功提示元素是否存在
        try:
            success_msg = WebDriverWait(self.driver, 10).until(
                EC.presence_of_element_located((By.CLASS_NAME, 'success-message'))
            )
            return True
        except:
            return False
    
    def close(self):
        """关闭浏览器"""
        if self.driver:
            self.driver.quit()


# 使用示例
if __name__ == '__main__':
    publisher = MultiPlatformPublisher('xianyu')
    publisher.start_browser()
    
    item = {
        'title': '二手 iPhone 12 99 新',
        'description': '使用一年，几乎全新',
        'price': 3500,
        'images': ['/path/to/image1.jpg', '/path/to/image2.jpg']
    }
    
    success = publisher.publish_to_xianyu(item)
    print(f"发布{'成功' if success else '失败'}")
    
    publisher.close()
```

#### **功能 2: 定时发布任务**

**实现原理**: 使用 APScheduler 调度器

**代码实现** (`src/scheduler.py`):
```python
#!/usr/bin/env python3
"""
定时任务调度器
"""

from apscheduler.schedulers.blocking import BlockingScheduler
from apscheduler.triggers.cron import CronTrigger
from datetime import datetime

class PublishScheduler:
    """发布调度器"""
    
    def __init__(self):
        self.scheduler = BlockingScheduler()
        self.tasks = []
    
    def add_task(self, task_name: str, publish_time: str, item: dict, platforms: list):
        """
        添加定时任务
        
        Args:
            task_name: 任务名称
            publish_time: 发布时间 (格式：HH:MM 或 cron 表达式)
            item: 商品信息
            platforms: 发布平台列表
        """
        # 解析发布时间
        if ':' in publish_time:
            # 简单时间格式 (HH:MM)
            hour, minute = publish_time.split(':')
            trigger = CronTrigger(hour=int(hour), minute=int(minute))
        else:
            # cron 表达式
            trigger = CronTrigger.from_crontab(publish_time)
        
        # 添加任务
        self.scheduler.add_job(
            func=self._execute_task,
            trigger=trigger,
            args=[task_name, item, platforms],
            id=task_name,
            replace_existing=True
        )
        
        self.tasks.append({
            'name': task_name,
            'time': publish_time,
            'item': item,
            'platforms': platforms
        })
        
        print(f"✅ 任务已添加：{task_name} - {publish_time}")
    
    def _execute_task(self, task_name: str, item: dict, platforms: list):
        """执行发布任务"""
        print(f"\n⏰ 执行任务：{task_name}")
        print(f"时间：{datetime.now()}")
        
        from src.publisher import MultiPlatformPublisher
        
        for platform in platforms:
            print(f"\n发布到：{platform}")
            
            publisher = MultiPlatformPublisher(platform)
            publisher.start_browser()
            
            try:
                if platform == 'xianyu':
                    success = publisher.publish_to_xianyu(item)
                elif platform == 'zhuanzhuan':
                    success = publisher.publish_to_zhuanzhuan(item)
                else:
                    success = False
                
                print(f"{'✅ 成功' if success else '❌ 失败'}")
                
            except Exception as e:
                print(f"❌ 异常：{e}")
            
            finally:
                publisher.close()
    
    def start(self):
        """启动调度器"""
        print("🚀 调度器启动...")
        print(f"当前任务数：{len(self.tasks)}")
        
        try:
            self.scheduler.start()
        except (KeyboardInterrupt, SystemExit):
            print("🛑 调度器停止")
    
    def list_tasks(self):
        """列出所有任务"""
        print("\n📋 当前任务列表:")
        print("-" * 60)
        for task in self.tasks:
            print(f"任务：{task['name']}")
            print(f"时间：{task['time']}")
            print(f"平台：{', '.join(task['platforms'])}")
            print(f"商品：{task['item'].get('title', 'N/A')}")
            print("-" * 60)


# 使用示例
if __name__ == '__main__':
    scheduler = PublishScheduler()
    
    # 添加定时任务
    scheduler.add_task(
        task_name='morning_post',
        publish_time='09:00',
        item={
            'title': '二手商品',
            'price': 100
        },
        platforms=['xianyu', 'zhuanzhuan']
    )
    
    # 启动调度器
    scheduler.start()
```

#### **功能 3: 多账号管理**

**实现方式**: 配置文件 + Cookie 管理

**代码实现** (`src/account_manager.py`):
```python
#!/usr/bin/env python3
"""
账号管理器
"""

import json
from pathlib import Path
from datetime import datetime

class AccountManager:
    """账号管理器"""
    
    def __init__(self, accounts_file: str = 'config/accounts.json'):
        self.accounts_file = Path(accounts_file)
        self.accounts = self._load_accounts()
    
    def _load_accounts(self) -> dict:
        """加载账号配置"""
        if self.accounts_file.exists():
            with open(self.accounts_file, 'r', encoding='utf-8') as f:
                return json.load(f)
        return {}
    
    def add_account(self, platform: str, account_name: str, credentials: dict):
        """
        添加账号
        
        Args:
            platform: 平台名称
            account_name: 账号名称
            credentials: 账号凭证（用户名/密码/Cookie）
        """
        if platform not in self.accounts:
            self.accounts[platform] = {}
        
        self.accounts[platform][account_name] = {
            'credentials': credentials,
            'created_at': datetime.now().isoformat(),
            'last_used': None,
            'status': 'active'
        }
        
        self._save_accounts()
        print(f"✅ 账号已添加：{platform}/{account_name}")
    
    def get_account(self, platform: str, account_name: str) -> dict:
        """获取账号信息"""
        platform_accounts = self.accounts.get(platform, {})
        account = platform_accounts.get(account_name)
        
        if account:
            # 更新最后使用时间
            account['last_used'] = datetime.now().isoformat()
            self._save_accounts()
        
        return account
    
    def list_accounts(self, platform: str = None) -> list:
        """列出所有账号"""
        if platform:
            return list(self.accounts.get(platform, {}).keys())
        else:
            all_accounts = []
            for plat, accounts in self.accounts.items():
                for name in accounts.keys():
                    all_accounts.append(f"{plat}/{name}")
            return all_accounts
    
    def _save_accounts(self):
        """保存账号配置"""
        self.accounts_file.parent.mkdir(parents=True, exist_ok=True)
        with open(self.accounts_file, 'w', encoding='utf-8') as f:
            json.dump(self.accounts, f, indent=2, ensure_ascii=False)


# 使用示例
if __name__ == '__main__':
    manager = AccountManager()
    
    # 添加账号
    manager.add_account(
        platform='xianyu',
        account_name='main_account',
        credentials={
            'username': 'your_username',
            'password': 'your_password',  # 建议使用加密存储
            'cookie': 'session_cookie'
        }
    )
    
    # 获取账号
    account = manager.get_account('xianyu', 'main_account')
    
    # 列出账号
    all_accounts = manager.list_accounts()
    print(f"所有账号：{all_accounts}")
```

## 7. 故障排除

### **7.1 常见问题**

**问题 1: ChromeDriver 版本不匹配**
```
❌ Message: ChromeDriver only supports Chrome version XX
```
**解决**:
```bash
# 检查 Chrome 版本
google-chrome --version

# 下载对应版本 ChromeDriver
# https://chromedriver.chromium.org/downloads
```

**问题 2: 登录状态失效**
```
❌ 需要重新登录
```
**解决**:
- 手动登录一次
- 保存 Cookie
- 使用 Cookie 登录

**问题 3: 发布失败**
```
❌ 发布失败：元素未找到
```
**解决**:
- 检查页面是否完全加载
- 增加等待时间
- 检查元素选择器是否正确

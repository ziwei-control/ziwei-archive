VALID_API_KEYS = {
    "user1": "sk-abc123...",
    "user2": "sk-def456...",
}

def verify_api_key(request):
    api_key = request.headers.get('X-API-Key')
    if api_key not in VALID_API_KEYS.values():
        return jsonify({"error": "Invalid API Key"}), 401
    return None
